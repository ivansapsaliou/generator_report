-- Статистика по нагрузке (с момента последнего сброса статистики)
SELECT 
    SUM(decode(stat_name, 'session logical reads', value, 0)) as logical_reads,
    SUM(decode(stat_name, 'physical reads', value, 0)) as physical_reads,
    SUM(decode(stat_name, 'physical writes', value, 0)) as physical_writes,
    SUM(decode(stat_name, 'user rollbacks', value, 0)) as rollbacks,
    SUM(decode(stat_name, 'user commits', value, 0)) as commits
FROM v$sysstat
WHERE stat_name IN (
    'session logical reads',
    'physical reads',
    'physical writes',
    'user rollbacks',
    'user commits',
    'parse count (total)',
    'execute count'
);
Buffer Cache Hit Ratio (аналог shared_buffers)
SELECT ROUND(
    (1 - (phy.value / (cur.value + con.value))) * 100, 2
) as buffer_cache_hit_ratio_percent
FROM v$sysstat cur, v$sysstat con, v$sysstat phy
WHERE cur.name = 'db block gets'
  AND con.name = 'consistent gets'
  AND phy.name = 'physical reads';

TOP 15 таблиц по размеру (как на вашем скриншоте)
SELECT * FROM (
    SELECT 
        owner,
        segment_name as table_name,
        ROUND(SUM(bytes)/1024/1024/1024, 2) as size_gb,
        ROUND(SUM(bytes)/1024/1024, 2) as size_mb,
        COUNT(*) as extents
    FROM dba_segments
    WHERE segment_type = 'TABLE'
    GROUP BY owner, segment_name
    ORDER BY SUM(bytes) DESC
) WHERE ROWNUM <= 15;

Статистика по таблицам (аналог pg_stat_user_tables)
SELECT 
    t.owner,
    t.table_name,
    t.num_rows,
    t.blocks,
    t.avg_row_len,
    t.last_analyzed,
    ROUND(t.num_rows * t.avg_row_len / 1024 / 1024, 2) as estimated_size_mb
FROM dba_tables t
WHERE t.owner NOT IN ('SYS', 'SYSTEM', 'ORDSYS', 'CTXSYS')
ORDER BY t.num_rows DESC
FETCH FIRST 20 ROWS ONLY;

🗄️ 5. TABLESPACE (Аналог диска в PostgreSQL)
Использование tablespace
SELECT 
    df.tablespace_name,
    ROUND(df.bytes / 1024 / 1024 / 1024, 2) as size_gb,
    ROUND((df.bytes - SUM(fs.bytes)) / 1024 / 1024 / 1024, 2) as used_gb,
    ROUND(SUM(fs.bytes) / 1024 / 1024 / 1024, 2) as free_gb,
    ROUND(((df.bytes - SUM(fs.bytes)) / df.bytes) * 100, 2) as used_percent
FROM dba_data_files df
LEFT JOIN dba_free_space fs ON df.tablespace_name = fs.tablespace_name
GROUP BY df.tablespace_name, df.bytes
ORDER BY used_percent DESC;

🔥 7. TOP SQL ЗАПРОСЫ (Опционально)
Самые ресурсоемкие запросы
SELECT * FROM (
    SELECT 
        sql_id,
        ROUND(elapsed_time/1000000, 2) as elapsed_sec,
        ROUND(cpu_time/1000000, 2) as cpu_sec,
        executions,
        ROUND(elapsed_time/1000000/nullif(executions,0), 2) as avg_elapsed_sec,
        SUBSTR(sql_text, 1, 100) as sql_preview
    FROM v$sqlarea
    WHERE executions > 0
    ORDER BY elapsed_time DESC
) WHERE ROWNUM <= 10;
